def extract_head!
	extraer=@head
	
	if(sz>0)
       @head= head.next
       
       if head.nil?
          @tail=nil
       else
          @head.prev=nil
       end
          
       extraer.next=nil
       extraer.prev=nil
	   @sz= @sz-1
	end
    extraer
end

def insert_head!(val)
	if(@sz==0)
	   @head=Node.new(val,nil,nil)
	   @tail= head
	   @sz= @sz+1
	else
	   @head.prev=Node.new(val,head,nil)
	   @head= head.prev
	   @sz= @sz+1
    end
end
