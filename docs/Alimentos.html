Overview

def aibc
   
   almacen=Array.new
   datos_IG.each do |x|
      almacen<<[]
      x.each_cons(2){ |val|  almacen.last<<((val[1]-x[0]+val[0]-x[0])/2)*5 } 
   end
   
   almacen.map{ |x| (x.reduce :+).round(2) }
end
